### To create this route:

## Folder structure:
```
app/
  category/
     [cId]/
        product/
           [pId]/
              page.tsx
```
Example: `page.tsx`
```
export default async function ProductCategory({ params }: {
    params: Promise<{ pId: string, cId: string }>
}) {
    // const cId = (await params).cId;
    // const pId = (await params).pId;
        
    const { cId, pId } = (await params)
    return <h1 className="text-5xl">Category : {cId} : Product : {pId}</h1>
}
```

- return type : `Resolve a Promise : ` `<[folderName] : type,[folderName] : type,......>`
- Resolve the promise -- `(await params)`
- `http://localhost:3000/category[cId]/product/[pId]
